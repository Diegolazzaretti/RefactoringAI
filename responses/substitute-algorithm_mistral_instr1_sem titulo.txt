def find_person(people):
    for name in people:
        if name == "Don":
            return "Don"
        elif name == "John":
            return "John"
        elif name == "Kent":
            return "Kent"
    return "" 

# Usage example
people = ["Don", "John", "Kent", "Bob"]
print(find_person(people)) # Output: Don
